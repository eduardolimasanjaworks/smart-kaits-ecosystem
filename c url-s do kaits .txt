# busca alunos
curl -X POST "https://api.kaits.com.br/alunos/" \
  -F acao=alunos \
  -F token=SEU_TOKEN_AQUI \
  -F filtro={filtro} \
  -F nome={nome} \
  -F licenca={licenca} \
  -F numMatric={numMatric} \
  -F CPF={CPF} \
  -F RG={RG} \
  -F email={email}
# Resposta:
#   - sucesso: sucesso da consulta (bit: "0" ou "1")
#   - msg: mensagem de erro (se sucesso = 0)
#   - alunos: uma lista com os alunos de acordo com os filtros com os seguintes campos:
#   - idAluno: identificador do aluno no Sistema KAITS
#   - nome: nome do aluno
#   - codinome: codinome (nome social) do aluno
#   - registro: registro do aluno
#   - nascimento: data de nascimento do aluno
#   - CPF: CPF do aluno
#   - RG: RG do aluno
#   - emails: e-mails cadastrados do aluno separados por ponto e vírgula (;)
#   - licencaIni: data de início da licença do aluno
#   - licencaFim: data de término da licença do aluno

# busca informações de um aluno
curl -X POST "https://api.kaits.com.br/alunos/" \
  -F acao=umAluno \
  -F token=SEU_TOKEN_AQUI \
  -F idAluno={idAluno}
# Resposta:
#   - sucesso: sucesso da consulta (bit: "0" ou "1")
#   - msg: mensagem de erro (se sucesso = 0)
#   - aluno: objeto com os dados do aluno
#   - idAluno: identificador do aluno no Sistema KAITS
#   - nome: nome do aluno
#   - codinome: codinome (nome social) do aluno
#   - registro: registro do aluno
#   - nascimento: data de nascimento do aluno
#   - licencaIni: data de início da licença do aluno
#   - licencaFim: data de término da licença do aluno
#   - perfil: nome do perfil do aluno
#   - obs: observação do aluno
#   - identidades: lista de documentos de identificação do aluno com:
#   - tipo: tipo da identificação
#   - numero: número da identificação
#   - emails: lista de e-mails do aluno com:
#   - tipo: tipo do e-mail
#   - email: e-mail
#   - fones: lista de telefones do aluno com:
#   - tipo: tipo do telefone
#   - ddi: DDI do telefone
#   - ddd: DDD do telefone
#   - fone: telefone
#   - enderecos: lista de endereços do aluno com:
#   - tipo: tipo do endereço
#   - rua: logradouro
#   - num: número
#   - compl: complemento
#   - bairro: bairro
#   - cidade: cidade
#   - UF: estado
#   - CEP: CEP
#   - matriculas: lista com as matrículas do aluno com:
#   - idMatric: identificador interno da matrícula
#   - numMatric: número da matrícula
#   - ini: data de início da matrícula
#   - fim: data de término da matrícula
#   - curso: curso da matrícula
#   - estagio: estágio da matrícula
#   - turma: turma da matrícula
#   - responsaveis: lista com os responsáveis pelo aluno com:
#   - idResp: identificador único do responsável
#   - nome: nome do responsável
#   - codinome: codinome (nome social) do responsável
#   - relacionamento: relacionamento entre o responsável e o aluno
#   - ehFinanceiro: se é o responsável financeiro do aluno
#   - ehPedagogico: se é o responsável pedagógico do aluno
#   - nascimento: data de nascimento do responsável
#   - e-mail: e-mail pessoal do responsável
#   - fone: telefone do responsável
#   - cel: celular do responsável
#   - RG: RG do responsável
#   - CPF: CPF do responsável

# cadastra aula
curl -X POST "https://api.kaits.com.br/aulas/" \
  -F acao=criaAula \
  -F token=SEU_TOKEN_AQUI \
  -F aula={aula} \
  -F turma={turma} \
  -F data={data} \
  -F hora={hora} \
  -F tempo={tempo} \
  -F curriculo={curriculo} \
  -F nome={nome} \
  -F professor={professor} \
  -F sala={sala} \
  -F modalidade={modalidade} \
  -F aulaRep={aulaRep} \
  -F remunera={remunera} \
  -F "C"={\"C\"} \
  -F "E"={\"E\"} \
  -F "T"={\"T\"} \
  -F "P"={\"P\"} \
  -F valor={valor} \
  -F remuneraTipo={remuneraTipo} \
  -F exemplo:={exemplo:}
# Resposta:
#   - sucesso: sucesso da gravação da aula (bit: "0" ou "1")
#   - idAula: identificador único da aula criada
#   - msg: array com mensagens de erro (se sucesso = 0)

# busca as aulas de uma turma

  -F acao=aulasTurma \
  -F token=SEU_TOKEN_AQUI \
  -F idTurm={idTurm}
# Resposta:
#   - sucesso: sucesso da consulta (bit: "0" ou "1")
#   - msg: mensagem de erro (se sucesso = 0)
#   - aulas: uma lista com as aulas da turma informada com os seguintes campos:
#   - idAula: ID da aula
#   - nome: nome da aula
#   - data: data da aula (AAAA-MM-DD)
#   - hora: hora do início da aula (hh:mm)
#   - tempo: duração da aula em minutos
#   - professor: ID do professor da aula (se houver mais de um professor na aula os IDs estarão separados por vírgula)
#   - profNome: nome do professor da aula e sua função se estiver gravada (se houver mais de um professor na aula os nomes estarão separados por vírgula)
#   - sala: ID da sala da aula
#   - salaNome: nome da sala da aula
#   - salaLink: URL para a sala da aula
#   - modalidade: ID da modalidade da aula
#   - modalidadeNome: nome da modalidade da aula
#   - status: status da aula (prevista, realizada ou cancelada)

# busca informações de uma aula
curl -X POST "https://api.kaits.com.br/aulas/" \
  -F acao=umaAula \
  -F token=SEU_TOKEN_AQUI \
  -F idAula={idAula}
# Resposta:
#   - sucesso: sucesso da consulta (bit: "0" ou "1")
#   - msg: mensagem de erro (se sucesso = 0)
#   - nome: nome da aula
#   - data: data da aula (AAAA-MM-DD)
#   - hora: hora do início da aula (hh:mm)
#   - tempo: duração da aula em minutos
#   - curso: ID do curso da aula
#   - cursoNome: nome do curso da aula
#   - estagio: ID do estágio da aula
#   - estagioNome: nome do estágio da aula
#   - turma: ID da turma da aula
#   - turmaNome: nome da turma da aula
#   - disciplina: ID da disciplina da aula
#   - discipNome: nome da disciplina da aula
#   - curriculo: currículo ministrado na aula
#   - professor: ID do professor da aula (se houver mais de um professor na aula os IDs estarão separados por vírgula)
#   - profNome: nome do professor da aula e sua função se estiver gravada (se houver mais de um professor na aula os nomes estarão separados por vírgula)
#   - sala: ID da sala da aula
#   - salaNome: nome da sala da aula
#   - salaLink: URL para a sala da aula
#   - modalidade: ID da modalidade da aula
#   - modalidadeNome: nome da modalidade da aula
#   - status: status da aula (prevista, realizada ou cancelada)
#   - aulaReposta: se for uma aula de reposição, informa o ID da aula reposta
#   - remuneracao: valor de remuneração do professor na aula
#   - remuneraTipo: tipo da remuneração do professor: "1" - por aula, "2" - por hora
#   - alunos: uma lista com as alunos que deveriam cursar a aula com os seguintes campos:
#   - idMatric: ID da matrícula do aluno que está na aula
#   - nome: nome do aluno
#   - iniMatric: início da matrícula do aluno nesta turma (AAAA-MM-DD)
#   - fimMatric: término da matrícula do aluno nesta turma (AAAA-MM-DD)
#   - pres: presença do aluno na aula (bit: 0 ou 1)
#   - tipoPres: tipo de presença do aluno na aula (se existirem tipos de presença cadastrados)
#   - just: ID da justificativa da ausência do aluno na aula (se não presente)
#   - justNome: texto da justificativa da ausência do aluno na aula
#   - obs: observação sobre o aluno na aula

#   - msg: mensagem de erro (se sucesso = 0)

#   - id: id do sistema de avaliação
#   - nome: nome do sistema de avaliação
#   - ativo: 1: ativo, 2: inativo
#   - aplicacao: a que o sistema de avaliação se refere: 1: notas, 2: testes, 3: boletim
#   - tipoNota: tipo de nota -> 1: numérico, 2: alfabético, 3: conceitos, 4: livre
#   - notaMin: nota mínima
#   - notaMax: nota máxima
#   - totalizacao: 0: sem totalizações, 1; com médias, 2: com totais
#   - precisao: variação das notas
#   - arredondamento: arredondamento dos totais
#   - considNotaEmBranco: considera notas em branco nas médias -> 0: não, 1: sim
#   - mediaParcial: exibe média ou total por parcial -> 0: não, 1: sim
#   - mediaTopico: exibe média ou total por tópico -> 0: não, 1: sim
#   - temPrincipal: tem nota principal -> 0: não, 1: sim
#   - principal: nota principal do sistema de avaliação
#   - parteBoletim: faz parte de um boletim -> 0: não, 1: sim

#   - topicos: lista com os tópicos do sistema de avaliação com:

#   - subtopicos: lista com subtopicos com:

#   - peso: peso do subtopico
#   - max: nota máxima do subtopico
#   - tipoNota: tipo de nota específico do subtopico (os tipos 5:"dados gravado" e 6:"nota calculada" são automáticos do sistema, não editáveis)
#   - precisao: precisão específica do subtopico
#   - parciais: lista com as parciais do sistema de avaliação com:
#   - id: id da parcial
#   - nome: nome da parcial
#   - peso: peso da parcial
#   - max: nota máxima da parcial
#   - ordem: ordem da parcial
#   - coment: parcial com comentários -> 0: não, 1: sim
#   - origem: origem da parcial -> 0: nota de outra avaliação, 1: nota a ser atribuída, 2: nota a ser calculada
#   - tipoCalculo: tipo de cálculo da parcial -> 0: média, 1: soma, 2: maior, 3: média desde que acima da marcada
#   - calcFinal: só calcula a parcial se todos parciais preenchidos -> 0: não, 1: sim
#   - exibeDoc: exibe a parcial no boletim -> 0: não, 1: sim
#   - abrev: abreviação da parcial mostrada no boletim
#   - parciaisCalc: lista das parciais de cálculo com:
#   - parcial: id da parcial de cálculo
#   - peso: peso da parcial de cálculo
#   - marcada: parcial de cálculo marcada (para limite do cálculo da média) -> 0: não, 1: sim
#   - sitFinal: lista com as situações finais do sistema de avaliação com:
#   - id: id da situação final
#   - nome: nome da situação final
#   - ordem: ordem da situação final
#   - conceitos: lista com os conceitos do sistema de avaliação com:
#   - id: id do conceito
